## ソースコード

``` js
/**
 * 正規表現を用いて、拡張子を得る
 *
 * @param { String } f
 */
export const getExt__Regex = (f) => {
  return f.split(/(?=\.[^.]+$)/)[1] || null
}

/**
 * Node関数を用いて、拡張子を得る
 *
 * @param { String } f
 */
export const getExt__Node = (f) => extname(f) || null
```
